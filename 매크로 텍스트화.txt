1. 원하는미주로이동

function OnScriptMacro_미주숫자로이동()
{
	A=GetPosBySet();
	ListA=A.item("List");
	ParaA=A.Item("Para");
	PosA=A.Item("Pos");

	MovePos(2,0,0); // 문서 제일 앞으로 이동
	미주번호();
	HAction.Run("MoveSelRight");
	var startnumber = GetTextFile("TEXT","saveblock");
	var startnumber =  startnumber.replace( /\./g, "" );
	var startnumber =  startnumber.replace( /\)/g, "" );

	MovePos(ListA,ParaA,PosA); // 원래 커서 위치로 이동

	 if ( startnumber == 1 ) {

	var Hc=HeadCtrl;
	cnt=0;
	do{
	if(Hc.CtrlID=="en"){	
		//setPosBySet(Hc.GetAnchorPos(0));
		cnt++;
}
	Hc=Hc.next
}while(Hc)

	str=prompt("이동할 미주 번호를 입력해 주세요.","코딩헤윰");

	var i = str;
	if ( cnt < i || i <= 0){
	with( XHwpMessageBox )
	{
	String =  i + "번은 없는 번호 입니다. 다시 입력해주세요.";
	Flag = 0;
	DoModal();
	}
	return;
	}

	if ( cnt >= i ){
	MovePos(2,0,0); // 문서 제일 앞으로 이동 

	미주번호();


	for(var k=1; k<i;k++){
	HAction.Run("NoteToNext"); //다음 미주 번호로 이동 
	}
	HAction.Run("CloseEx"); // 미주 번호 빠져나오기


	//화면 스크롤 다운 15번 
	HAction.Run("MoveScrollDown");
	HAction.Run("MoveScrollDown");
	HAction.Run("MoveScrollDown");
	HAction.Run("MoveScrollDown");
	HAction.Run("MoveScrollDown");
	HAction.Run("MoveScrollDown");
	HAction.Run("MoveScrollDown");
	HAction.Run("MoveScrollDown");
	HAction.Run("MoveScrollDown");


	}

	}
	else {

	MovePos(2,0,0); // 문서 제일 앞으로 이동
	var Hc=HeadCtrl;
	cnt=0;
	do{
	if(Hc.CtrlID=="en"){	
		//setPosBySet(Hc.GetAnchorPos(0));
		cnt++;
}
	Hc=Hc.next
}while(Hc)

	MovePos(ListA,ParaA,PosA); // 원래 커서 위치로 이동

	str=prompt("이동할 미주 번호를 입력해 주세요.","코딩헤윰");
	if ( str == null ) {
	 return;
	}
	else if(str != parseInt(str)) {
	with( XHwpMessageBox )
	{
	String =  "잘못된 번호 형식입니다. 다시 입력해주세요.";
	Flag = 0;
	DoModal();
	}
	return;
	}

	str=str.replace( /\s/g, "" );
	var arr=eval("["+str+"]");

	if (  arr < parseInt(startnumber) || parseInt(startnumber)+cnt-1 < arr) {
	with( XHwpMessageBox )
	{
	String =  arr + "번은 없는 번호 입니다. 다시 입력해주세요.";
	Flag = 0;
	DoModal();
	}
	return;
	}

	else {
	MovePos(2,0,0); // 문서 제일 앞으로 이동 

	미주번호();

	for(var k=0;k<arr-startnumber;k++){
	HAction.Run("NoteToNext"); //다음 미주 번호로 이동 
	}
	HAction.Run("CloseEx"); // 미주 번호 빠져나오기


	//화면 스크롤 다운 15번 
	HAction.Run("MoveScrollDown");
	HAction.Run("MoveScrollDown");
	HAction.Run("MoveScrollDown");
	HAction.Run("MoveScrollDown");
	HAction.Run("MoveScrollDown");
	HAction.Run("MoveScrollDown");
	HAction.Run("MoveScrollDown");
	HAction.Run("MoveScrollDown");
	HAction.Run("MoveScrollDown");

	}

}

function prompt(prompt, title) {
	if ( prompt === undefined ) prompt = '';
	if ( title === undefined ) tilte = '';
	var script = new ActiveXObject('MSScriptControl.ScriptControl');
	script.language = 'vbscript';
	return script.eval('inputbox("' + prompt + '","' + title + '")');
}

function 미주번호()

 act.GetDefault(set);
 set.SetItem("SetSelectionIndex", 5);
 set.SetItem("DialogResult", 32);
 ret=act.Execute(set);
}

function OnScriptMacro_박스서식수정용()
{
	FindCtrl();

	{

		WidthRelTo = WidthRel("Column");

		//Height = 7034;
		Width = 10000;
		ShapeDrawLineAttr.Color = RGBColor(0, 0, 0);

		OutsideMarginTop = MiliToHwpUnit(0.0);
		OutsideMarginRight = MiliToHwpUnit(0.2);
		OutsideMarginLeft = MiliToHwpUnit(0.2);

		ShapeListProperites.MarginBottom = MiliToHwpUnit(3.0);
		ShapeListProperites.MarginTop = MiliToHwpUnit(4.0);
		ShapeListProperites.MarginRight = MiliToHwpUnit(3.0);
		ShapeListProperites.MarginLeft = MiliToHwpUnit(3.0);
		ShapeListProperites.VertAlign = VAlign("Center");
	}
	HAction.Execute("ShapeObjDialog", HParameterSet.HShapeObject.HSet);
	HAction.Run("Cancel");
	HAction.Run("MoveLeft");

###31. 예전부등식수정용

function OnScriptMacro_예전부등식교체v2()
{

	// 문서 제일 앞으로 캐럿 이동 
	MovePos(2,0,0);

	code2=HeadCtrl;

	cnt=0;

	do
	{
		// 문서 전체의 수식을 순회하는 코드
		if(code2.CtrlID=="eqed")
		{
			// 수식으로 캐럿 이동 
			//SetPosBySet(code2.GetAnchorPos(0));

			// 수식으로 이동
			HAction.GetDefault("Goto", HParameterSet.HGotoE.HSet);
			with (HParameterSet.HGotoE)
			{
				HSet.SetItem("DialogResult", 37);
				SetSelectionIndex = 5;
			}
			HAction.Execute("Goto", HParameterSet.HGotoE.HSet);

			// 수식을 개체 지정 
			FindCtrl();

			// 부등식 변경 
			HAction.GetDefault("EquationPropertyDialog", HParameterSet.HShapeObject.HSet);
			with (HParameterSet.HShapeObject)
			{
				Width = 1012;
				HSet.SetItem("ShapeType", 3);
				String = String.replace(/≦/g, " LEQ " ); 
				String = String.replace(/<=/g, " LEQ " ); 
				String = String.replace(/≧/g, " GEQ " ); 
				String = String.replace(/>=/g, " GEQ " ); 		
			}
			HAction.Execute("EquationPropertyDialog", HParameterSet.HShapeObject.HSet);

			HAction.Run("Cancel");

			HAction.Run("MoveRight");

			cnt++;

		} // if 문 종료 
		code2=code2.Next;  

	}while(code2)

	HAction.Run("Cancel");
	HAction.Run("Cancel");
	MovePos(2,0,0);

	// 메세지 박스 
	XHwpMessageBox.String= " "+cnt+" 개의 수식을 처리하였습니다." ;
	XHwpMessageBox.Flag=0;
	XHwpMessageBox.DoModal();

}
